# 安装最新 master 分支
pip install git+https://github.com/henryczq/zzgz-autoto-core.git@master

# 更新到最新版本
pip install --upgrade git+https://github.com/henryczq/zzgz-autoto-core.git@master

# 强制重新安装
pip install --force-reinstall --no-cache-dir git+https://github.com/henryczq/zzgz-autoto-core.git@master
